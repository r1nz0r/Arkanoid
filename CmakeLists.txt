#Base Cmake config

cmake_minimum_required(VERSION 3.17)

project(Arkanoid
        VERSION 0.0.1
        LANGUAGES CXX
)

set(CMAKE_CXX_STANDARD          17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS        OFF)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
option(BUILD_SHARED_LIBS "Build shared libraries" OFF)

#Internal variables
set(ARKANOID_ENGINE_TARGET_NAME ArkanoidEngine)
set(ARKANOID_GAME_TARGET_NAME ArkanoidGame)
# Add engine and game Cmake files
add_subdirectory(${ARKANOID_ENGINE_TARGET_NAME})
